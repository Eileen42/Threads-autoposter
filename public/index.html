Apps Script 웹훅 방식으로 연동합니다. Google Cloud 계정·OAuth 설정 없이 5분 안에 완료됩니다.

Google Apps Script 열기
2
아래 코드 전체를 복사해서 붙여넣기 후 저장
function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var rows = data.rows;
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['ID','스크랩일시','프로젝트','계정','원본계정','원본URL','원문본문','첫댓글','미디어URL']);
  }
  rows.forEach(function(r) {
    sheet.appendRow([r.id, r.scraped_at, r.project_name, r.username, r.source_username, r.source_url, r.text_content, r.first_comment, r.media_urls]);
  });
  return ContentService.createTextOutput(JSON.stringify({ok:true,count:rows.length})).setMimeType(ContentService.MimeType.JSON);
}
